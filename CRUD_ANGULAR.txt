		    return this.http.get<Empleado[]>(`${this.apiURL}`); 
		}

		add(modelo:Empleado):Observable<Empleado>{
		    return this.http.post<Empleado>(`${this.apiURL}`, modelo);
		}

  		update(idEmpleado:number ,modelo:Empleado):Observable<Empleado>{
    			return this.http.post<Empleado>(`${this.apiURL}/${idEmpleado}`, modelo);
  		}

  		delete(idEmpleado:number): Observable<void>{
	    		return this.http.delete<void>(`${this.apiURL}/${idEmpleado}`);
  		}
